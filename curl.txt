### POST
curl -X POST http://localhost:8080/accounts \
     -H "Content-Type: application/json" \
     -d '{"owner": "Sasha", "currency": "RUB"}'




### GET
curl -X GET http://localhost:8080/accounts/5




### GET QUERY
curl -X GET http://localhost:8080/accounts?page_id=1&page_size=6




### POST TRANSFERS
curl -X POST http://localhost:8080/transfers \
     -H "Content-Type: application/json" \
     -d '{"from_account_id": 216, "to_account_id": 217, "amount": 100, "currency": "USD"}'
